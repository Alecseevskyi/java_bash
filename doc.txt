#include <iostream>
#include <vector>
#include <limits>

using namespace std;

// Функция для нахождения минимума в массиве (изменено с максимума)
double find_min(const vector<double>& numbers) {
    if (numbers.empty()) {
        return numeric_limits<double>::quiet_NaN();
    }
    
    double min_num = numbers[0];
    for (double num : numbers) {
        if (num < min_num) {  // Изменено: > на <
            min_num = num;
        }
    }
    return min_num;
}

int main() {
    int n;
    cout << "Введите количество элементов в массиве: ";
    cin >> n;
    
    if (n <= 0) {
        cout << "Количество элементов должно быть положительным числом!" << endl;
        return 1;
    }
    
    vector<double> numbers(n);
    
    for (int i = 0; i < n; ++i) {
        cout << "Введите элемент " << i + 1 << ": ";
        cin >> numbers[i];
    }
    
    double minimum = find_min(numbers);  // Изменено имя переменной
    
    if (isnan(minimum)) {
        cout << "Массив пуст!" << endl;
    } else {
        cout << "Минимальный элемент в массиве: " << minimum << endl;  // Изменено сообщение
    }
    
    return 0;
}
